68      private static Properties PROPS;
69      static {
70          // an up to date install.properties should be provided by the buildfile
71          String path = "resources/install.properties";
72          InputStream in =
73                  OptionDefinition.class.getClassLoader().getResourceAsStream(
74                          path);
75          PROPS = new Properties();
76          try {
77              PROPS.load(in);
78          } catch (Exception e) {
79              System.err.println("ERROR: Unable to load required resource: " +
80                      path);
81              System.exit(1);
82          }
83          TOMCAT = PROPS.getProperty("install.tomcat");
84          JDBC_DERBY = PROPS.getProperty("install.jdbc.derby");
85          JDBC_DERBY_NETWORK =
86                  PROPS.getProperty("install.jdbc.derbynetworkclient");
87          JDBC_MYSQL = PROPS.getProperty("install.jdbc.mysql");
88          JDBC_POSTGRESQL = PROPS.getProperty("install.jdbc.postgresql");
89          TOMCAT_BASENAME = PROPS.getProperty("install.tomcat.basename");
90          COMMONS_COLLECTIONS = PROPS.getProperty("install.commons.collections");
91          COMMONS_DBCP = PROPS.getProperty("install.commons.dbcp");
92          COMMONS_POOL = PROPS.getProperty("install.commons.pool");
93      }
