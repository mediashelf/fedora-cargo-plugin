49       * The key under which the container instance is stored in the plugin context. We store it so
50       * that it's possible to get back the same container instance even if this mojo is called in a

104     protected LocalContainer getContainer() throws MojoExecutionException {
105         LocalContainer container = null;
106         @SuppressWarnings("unchecked")
107         Map<Object, Object> context = getPluginContext();
108 
109         String containerKey = CONTEXT_CONTAINER_KEY_PREFIX + "." + containerId;
110 
111         if (context != null) {
112             container = (LocalContainer) context.get(containerKey);
113         }
114 
115         if (container == null) {
116             container = createNewContainer();
117         }
118 
119         if (context != null) {
120             context.put(containerKey, container);
121         }
122 
123         return container;
124     }
125 
126     private LocalContainer createNewContainer() throws MojoExecutionException {
127         String home = installContainer();
128         return configureContainer(home);
129     }
130 
131     protected String installContainer() throws MojoExecutionException {
132         File buildDir =
133                 new File(mavenProject.getBuild().getDirectory(), "extracts");
134         ArtifactResult result = getArtifact(containerArtifact);
135         try {
136             unzip(result.getArtifact().getFile(), buildDir);
137         } catch (FileNotFoundException e) {
138             throw new MojoExecutionException(e.getMessage(), e);
139         } catch (IOException e) {
140             throw new MojoExecutionException(e.getMessage(), e);
141         }
142 
143         // Some potentially unwarranted assumptions to determine the 
144         // install directory:
145         // if buildDir contains only one file, we assume that the child is the 
146         // home dir, otherwise we assume the archive did not contain a root dir 
147         // and we just return the buildDir
148         File[] subdirs = buildDir.listFiles();
149         if (subdirs != null && subdirs.length == 1) {
150             return subdirs[0].getAbsolutePath();
151         } else {
152             return buildDir.getAbsolutePath();
153         }
154     }

156     private InstalledLocalContainer configureContainer(String home)
157             throws MojoExecutionException {
158         getLog().info("Using container home: " + home);
159 
160         ConfigurationFactory configurationFactory =
161                 new DefaultConfigurationFactory();
162         StandaloneLocalConfiguration configuration =
163                 (StandaloneLocalConfiguration) configurationFactory
164                         .createConfiguration(
165                         containerId, ContainerType.INSTALLED,
166                         ConfigurationType.STANDALONE);
167         configuration.setProperty(ServletPropertySet.PORT, fedoraPort);
168 
169         DefaultContainerFactory cf = new DefaultContainerFactory();
170         InstalledLocalContainer c =
171                 (InstalledLocalContainer) cf.createContainer(containerId,
172                         ContainerType.INSTALLED, configuration);
173         c.setHome(home);
174         c.setOutput(containerLog);
175         c.setSystemProperties(getSystemProperties());
176 
177         deploy(c.getConfiguration());
178         return c;
179     }
180 
181     //FIXME: this currently deploys fedoraWar unless the current artifact
182     // is itself a war. We really should be more flexible here.
183     private void deploy(LocalConfiguration cfg) throws MojoExecutionException {
184         WAR war = null;
185         if (mavenProject.getPackaging().equalsIgnoreCase("war")) {
186             getLog().info("Deploying " + mavenProject.getArtifact().getFile()
187                                     .getAbsolutePath());
188             war = new WAR(mavenProject.getArtifact().getFile()
189                             .getAbsolutePath());
190         } else {
191             war =
192                     new WAR(getArtifact(fedoraWar).getArtifact().getFile()
193                             .getAbsolutePath());
194         }
195         war.setContext("fedora");
196         cfg.addDeployable(war);
197     }

199     /**
200      * Ensures that fedora.home is set as a system prop (if it was defined)
201      * @return
202      */
203     private Map<String, String> getSystemProperties() {
204         if (systemProperties == null) {
205             systemProperties = new HashMap<String, String>();
206         }
207         if (!systemProperties.containsKey("fedora.home") &&
208                 fedoraHomeDir != null) {
209             systemProperties
210                     .put("fedora.home", fedoraHomeDir.getAbsolutePath());
211         }
212 
213         return systemProperties;
214     }
